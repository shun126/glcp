function(glcp_write_function_inventory source output)
  file(STRINGS "${source}" lines)
  set(content "/* generated by CMake from ${source} */\n")
  set(in_version_block OFF)
  set(version_guard "")
  set(version_content "")
  foreach(line IN LISTS lines)
    if(line MATCHES "^#if .*defined\\(GLCP_GL_VERSION_[0-9]+_[0-9]+\\)")
      set(in_version_block ON)
      set(version_guard "${line}")
      set(version_content "")
    elseif(in_version_block AND line MATCHES "^#endif$")
      if(NOT version_content STREQUAL "")
        string(APPEND content "${version_guard}\n")
        string(APPEND content "${version_content}")
        string(APPEND content "#endif\n")
      endif()
      set(in_version_block OFF)
    elseif(in_version_block AND line MATCHES "^[ \t]*(gl[A-Za-z0-9_]+) = \\(PFNGL[A-Za-z0-9_]+PROC\\)glcpGetProcAddress\\(\"gl[A-Za-z0-9_]+\"\\)")
      string(REGEX REPLACE "^[ \t]*(gl[A-Za-z0-9_]+) = .*" "\\1" function_name "${line}")
      string(APPEND version_content "GLCP_TEST_FUNCTION(${function_name})\n")
    endif()
  endforeach()
  file(WRITE "${output}" "${content}")
endfunction()

function(glcp_write_versioned_function_inventory source output)
  file(STRINGS "${source}" lines)
  set(content "/* generated by CMake from ${source} */\n")
  set(in_version_block OFF)
  set(version_guard "")
  set(version_name "")
  set(version_content "")
  foreach(line IN LISTS lines)
    if(line MATCHES "^#if .*defined\\((GLCP_GL_VERSION_[0-9]+_[0-9]+)\\)")
      set(in_version_block ON)
      set(version_guard "${line}")
      set(version_name "${CMAKE_MATCH_1}")
      set(version_content "")
    elseif(in_version_block AND line MATCHES "^#endif$")
      if(NOT version_content STREQUAL "")
        string(APPEND content "${version_guard}\n")
        string(APPEND content "${version_content}")
        string(APPEND content "#endif\n")
      endif()
      set(in_version_block OFF)
      set(version_guard "")
      set(version_name "")
    elseif(in_version_block AND line MATCHES "^[ \t]*(gl[A-Za-z0-9_]+) = \\(PFNGL[A-Za-z0-9_]+PROC\\)glcpGetProcAddress\\(\"gl[A-Za-z0-9_]+\"\\)")
      string(REGEX REPLACE "^[ \t]*(gl[A-Za-z0-9_]+) = .*" "\\1" function_name "${line}")
      string(APPEND version_content "GLCP_TEST_VERSION_FUNCTION(${version_name}, ${function_name})\n")
    endif()
  endforeach()
  file(WRITE "${output}" "${content}")
endfunction()

function(glcp_apply_common_test_settings target)
  target_include_directories(${target} PRIVATE "${CMAKE_SOURCE_DIR}" "${CMAKE_BINARY_DIR}")
  if(APPLE)
    target_compile_options(${target} PRIVATE -Wno-deprecated-declarations)
  endif()
endfunction()

function(glcp_add_compile_smoke target source)
  add_executable(${target} "${source}")
  glcp_apply_common_test_settings(${target})
  add_test(NAME ${target} COMMAND ${target})
endfunction()

function(glcp_add_runtime_smoke target source loader_source)
  add_executable(${target} "${source}" "${loader_source}" ${ARGN})
  glcp_apply_common_test_settings(${target})
  if(WIN32)
    target_link_libraries(${target} PRIVATE opengl32 user32 gdi32)
  elseif(APPLE)
    target_link_libraries(${target} PRIVATE "-framework OpenGL")
  else()
    target_link_libraries(${target} PRIVATE X11 GL dl)
  endif()
  add_test(NAME ${target} COMMAND ${target})
endfunction()

function(glcp_add_generated_validation_test target label header source inventory versioned_inventory)
  add_test(
    NAME ${target}
    COMMAND
      ${Python3_EXECUTABLE}
      "${CMAKE_SOURCE_DIR}/tests/validate_generated_loader.py"
      --label "${label}"
      --header "${header}"
      --source "${source}"
      --inventory "${inventory}"
      --versioned-inventory "${versioned_inventory}"
  )
endfunction()
